# I will provide you with an enhanced version of the clustering visualization script.
# This script is written to be used in your local environment since I cannot execute it here.

import pandas as pd
from sklearn.preprocessing import StandardScaler
from sklearn.cluster import KMeans
import matplotlib.pyplot as plt
import warnings

# Suppress FutureWarnings from scikit-learn
warnings.simplefilter(action='ignore', category=FutureWarning)

# Load the dataset
file_path = 'online_shoppers_intention.csv'  # Make sure this path is correct
data = pd.read_csv(file_path)

# Assume data preprocessing is already done here (e.g., missing value imputation)

# Select features for clustering, for example, 'ProductRelated_Duration' and 'BounceRates'
features = data[['ProductRelated_Duration', 'BounceRates']]

# Standardize the features to have a mean of zero and a standard deviation of one
scaler = StandardScaler()
X_scaled = scaler.fit_transform(features)

# Fit the KMeans model with an appropriate number of clusters
# You might want to use the elbow method or other techniques to find the optimal number of clusters
n_clusters = 3  # Example number of clusters
km = KMeans(n_clusters=n_clusters, init='k-means++', max_iter=300, n_init=10, random_state=0)
y_means = km.fit_predict(X_scaled)

# Plotting the clusters with enhanced visuals
plt.figure(figsize=(12, 8))  # Larger figure size for better visibility

# Scatter plot for clusters
colors = ['yellow', 'pink', 'cyan']  # Colors for the clusters
labels = ['Cluster 1', 'Cluster 2', 'Cluster 3']  # Labels for the clusters
for i in range(n_clusters):
    plt.scatter(X_scaled[y_means == i, 0], X_scaled[y_means == i, 1], s=100, c=colors[i], label=f'{labels[i]}')

# Scatter plot for centroids
plt.scatter(km.cluster_centers_[:, 0], km.cluster_centers_[:, 1], s=300, c='red', label='Centroids', marker='X')

# Enhance plot aesthetics
plt.title('Cluster Analysis of Shoppers', fontsize=22, fontweight='bold')
plt.xlabel('ProductRelated Duration (Standardized)', fontsize=16)
plt.ylabel('Bounce Rates (Standardized)', fontsize=16)
plt.xticks(fontsize=12)
plt.yticks(fontsize=12)
plt.legend(loc='best', fontsize=12)
plt.grid(True, which='both', linestyle='--', linewidth=0.5)

# Show the plot
plt.show()

# Add the cluster labels to the original dataframe
data['Cluster'] = y_means

# Optionally, save the DataFrame with cluster labels to a new CSV file
data.to_csv('online_shoppers_intention_with_clusters.csv', index=False)

# Print the first few entries of the data with cluster labels
print(data.head())
